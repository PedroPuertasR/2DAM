# Copiar la carpeta "pract03PHP" a "Documentos" e ir a ella:
cd /home/alumno/Documentos/pract03PHP

# En los ordenadores no restaurados hay que arrancar docker.
systemctl start docker

# Arrancar los tres contenedores con docker-compose. Debemos estar en la carpeta con el fichero "docker-compose.yml"
# La primera vez los descarga
docker-compose up -d

	Creating network "ejemplo_default" with the default driver
	Creating mysql
	Creating phpmyadmin
	Creating webserver

# En el navegador:
http://localhost/
http://localhost/alta.php

http://localhost/verDatos.php

# phpMyAdmin (usuario/1234)
http://localhost:8080/

# Parar contenedores:
docker-compose stop
	Stopping phpmyadmin ... done
	Stopping webserver ... done
	Stopping mysql ... done

# Parar y borrar contenedores:
docker-compose down
	Removing phpmyadmin ... done
	Removing webserver ... done
	Removing mysql ... done
	Removing network ejemplo_default
